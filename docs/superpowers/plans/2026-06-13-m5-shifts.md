// services/shift.rs

/// Kết quả tính đối soát ca — pure, không chạm DB.
pub struct ReconciliationResult {
    pub expected_cash: i64,
    pub cash_diff: i64,
    pub total_sales: i64,
}

/// Tính đối soát ca:
/// - expected_cash  = opening_cash + Σ cash_payments (payments.amount WHERE method='CASH',
///                    thuộc đơn PAID trong ca này)
/// - cash_diff      = closing_cash_counted − expected_cash
/// - total_sales    = Σ orders.total WHERE status='PAID' trong ca này
///
/// Tất cả dữ liệu đã được repo lấy ra và truyền vào dưới dạng slice.
pub fn compute_reconciliation(
    opening_cash: i64,
    closing_cash_counted: i64,
    cash_payment_amounts: &[i64],  // Σ payments.amount WHERE method='CASH' trong ca
    paid_order_totals: &[i64],     // orders.total WHERE status='PAID' trong ca
) -> ReconciliationResult {
    let cash_sum: i64 = cash_payment_amounts.iter().sum();
    let expected_cash = opening_cash + cash_sum;
    let cash_diff = closing_cash_counted - expected_cash;
    let total_sales: i64 = paid_order_totals.iter().sum();
    ReconciliationResult { expected_cash, cash_diff, total_sales }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn reconciliation_no_sales() {
        // Ca trống — không bán được gì, nhân viên đếm đủ tiền đầu
        let r = compute_reconciliation(500_000, 500_000, &[], &[]);
        assert_eq!(r.expected_cash, 500_000);
        assert_eq!(r.cash_diff, 0);
        assert_eq!(r.total_sales, 0);
    }

    #[test]
    fn reconciliation_cash_only() {
        // Mở ca 200_000; 3 đơn CASH: 50_000 + 80_000 + 70_000 = 200_000
        // expected = 200_000 + 200_000 = 400_000
        // Đếm thực 400_000 → diff = 0
        let r = compute_reconciliation(
            200_000,
            400_000,
            &[50_000, 80_000, 70_000],
            &[50_000, 80_000, 70_000],
        );
        assert_eq!(r.expected_cash, 400_000);
        assert_eq!(r.cash_diff, 0);
        assert_eq!(r.total_sales, 200_000);
    }

    #[test]
    fn reconciliation_mixed_payments_only_cash_counted() {
        // Mở ca 100_000; bán 3 đơn:
        //   đơn A: 150_000 (CASH) → cash_payment = 150_000
        //   đơn B: 200_000 (QR)   → cash_payment = 0 (không cộng vào tiền mặt)
        //   đơn C: 120_000 (CASH) → cash_payment = 120_000
        // expected = 100_000 + 150_000 + 120_000 = 370_000
        // total_sales = 150_000 + 200_000 + 120_000 = 470_000
        // Đếm thực 370_000 → diff = 0
        let r = compute_reconciliation(
            100_000,
            370_000,
            &[150_000, 120_000],           // chỉ CASH payments
            &[150_000, 200_000, 120_000],  // tất cả đơn PAID
        );
        assert_eq!(r.expected_cash, 370_000);
        assert_eq!(r.cash_diff, 0);
        assert_eq!(r.total_sales, 470_000);
    }

    #[test]
    fn reconciliation_shortage() {
        // Đếm thực thiếu 20_000
        let r = compute_reconciliation(100_000, 280_000, &[200_000], &[200_000]);
        assert_eq!(r.expected_cash, 300_000);
        assert_eq!(r.cash_diff, -20_000); // thiếu → âm
        assert_eq!(r.total_sales, 200_000);
    }

    #[test]
    fn reconciliation_surplus() {
        // Đếm thực thừa 10_000
        let r = compute_reconciliation(100_000, 310_000, &[200_000], &[200_000]);
        assert_eq!(r.expected_cash, 300_000);
        assert_eq!(r.cash_diff, 10_000); // thừa → dương
        assert_eq!(r.total_sales, 200_000);
    }

    #[test]
    fn reconciliation_negative_cash_diff_large_opening() {
        // Ca mở với tiền đầu lớn, không bán được đơn CASH nào, đếm thực thiếu
        let r = compute_reconciliation(1_000_000, 950_000, &[], &[]);
        assert_eq!(r.expected_cash, 1_000_000);
        assert_eq!(r.cash_diff, -50_000);
        assert_eq!(r.total_sales, 0);
    }

    #[test]
    fn reconciliation_multiple_cash_payments_same_order() {
        // Một đơn thanh toán CASH nhiều lần (split) — mỗi payments row được cộng riêng
        // opening = 0; payments CASH: 100_000 + 50_000 = 150_000; expected = 150_000
        let r = compute_reconciliation(0, 150_000, &[100_000, 50_000], &[150_000]);
        assert_eq!(r.expected_cash, 150_000);
        assert_eq!(r.cash_diff, 0);
        assert_eq!(r.total_sales, 150_000);
    }
}

/// Đóng ca:
/// 1. Validate ca tồn tại và OPEN.
/// 2. Query Σ payments.amount WHERE method='CASH' thuộc đơn PAID của ca này.
/// 3. Query Σ orders.total WHERE status='PAID' AND shift_id = shift_id.
/// 4. Tính reconciliation qua services/shift.
/// 5. UPDATE shifts row với kết quả.
pub fn close_shift(conn: &Connection, shift_id: i64, input: CloseShiftInput) -> rusqlite::Result<Shift> {
    // Validate ca tồn tại và OPEN
    let current_status: String = conn.query_row(
        "SELECT status FROM shifts WHERE id=?1",
        [shift_id],
        |r| r.get(0),
    )?;
    if current_status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName(
            "Ca phải đang OPEN mới đóng được".into(),
        ));
    }
    if input.closing_cash_counted < 0 {
        return Err(rusqlite::Error::InvalidParameterName(
            "closing_cash_counted không được âm".into(),
        ));
    }

    let opening_cash: i64 = conn.query_row(
        "SELECT opening_cash FROM shifts WHERE id=?1",
        [shift_id],
        |r| r.get(0),
    )?;

    // Σ payments.amount WHERE method='CASH' thuộc đơn PAID trong ca
    // Query: JOIN payments → orders, lọc shift_id và status='PAID' và method='CASH'
    let cash_payments: Vec<i64> = {
        let mut stmt = conn.prepare(
            "SELECT p.amount
             FROM payments p
             JOIN orders o ON o.id = p.order_id
             WHERE o.shift_id = ?1
               AND o.status = 'PAID'
               AND p.method = 'CASH'",
        )?;
        stmt.query_map([shift_id], |r| r.get(0))?
            .collect::<rusqlite::Result<_>>()?
    };

    // Σ orders.total WHERE status='PAID' AND shift_id = shift_id
    let paid_order_totals: Vec<i64> = {
        let mut stmt = conn.prepare(
            "SELECT total FROM orders WHERE shift_id=?1 AND status='PAID'",
        )?;
        stmt.query_map([shift_id], |r| r.get(0))?
            .collect::<rusqlite::Result<_>>()?
    };

    // Tính qua service (pure)
    let recon = svc::compute_reconciliation(
        opening_cash,
        input.closing_cash_counted,
        &cash_payments,
        &paid_order_totals,
    );

    use crate::services::order::now_utc;
    let closed_at = now_utc();

    conn.execute(
        "UPDATE shifts SET
             status               = 'CLOSED',
             closed_at            = ?2,
             closing_cash_counted = ?3,
             expected_cash        = ?4,
             cash_diff            = ?5,
             total_sales          = ?6,
             note                 = COALESCE(?7, note)
         WHERE id = ?1",
        params![
            shift_id,
            closed_at,
            input.closing_cash_counted,
            recon.expected_cash,
            recon.cash_diff,
            recon.total_sales,
            input.note,
        ],
    )?;

    load_shift(conn, shift_id)
}

- [ ] **Step 4: Tạo `src/routes/Shift.tsx`** — màn hình `/shift`: nếu không có ca → hiện `OpenShiftForm`; nếu có ca OPEN → hiện thông tin ca + `CloseShiftForm`:

```tsx
import { useEffect, useState } from "react"
import { Link } from "react-router-dom"
import { Button } from "@/components/ui/button"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { getCurrentShift, type Shift } from "@/lib/api/shifts"
import OpenShiftForm from "@/routes/shift/OpenShiftForm"
import CloseShiftForm from "@/routes/shift/CloseShiftForm"
import { formatVnd } from "@/lib/format"

export default function ShiftScreen() {
  const [shift, setShift] = useState<Shift | null | undefined>(undefined) // undefined = loading
  const [error, setError] = useState<string | null>(null)

  async function refresh() {
    try {
      const current = await getCurrentShift()
      setShift(current)
    } catch (e) {
      setError(String(e))
    }
  }

  useEffect(() => { refresh() }, [])

  if (shift === undefined) {
    return <div className="p-6 text-muted-foreground">Đang tải...</div>
  }

  return (
    <div className="mx-auto flex max-w-lg flex-col gap-4 p-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-bold">Ca làm việc</h1>
        <Button variant="outline" asChild>
          <Link to="/">Về trang chủ</Link>
        </Button>
      </div>

      {error && <p className="text-sm text-destructive">{error}</p>}

      {shift === null ? (
        <Card>
          <CardHeader>
            <CardTitle>Chưa có ca nào đang mở</CardTitle>
          </CardHeader>
          <CardContent>
            <OpenShiftForm onOpened={(s) => setShift(s)} />
          </CardContent>
        </Card>
      ) : (
        <>
          <Card>
            <CardHeader>
              <CardTitle className="flex items-center gap-2">
                <span className="inline-block h-2 w-2 rounded-full bg-green-500" />
                Ca đang mở
              </CardTitle>
            </CardHeader>
            <CardContent className="flex flex-col gap-1 text-sm">
              <span>Mở lúc: <strong>{new Date(shift.opened_at).toLocaleString("vi-VN")}</strong></span>
              <span>Tiền đầu: <strong>{formatVnd(shift.opening_cash)}</strong></span>
              {shift.note && <span>Ghi chú: {shift.note}</span>}
            </CardContent>
          </Card>
          <Card>
            <CardContent className="pt-6">
              <CloseShiftForm
                shift={shift}
                onClosed={(closed) => setShift(null)}
              />
            </CardContent>
          </Card>
        </>
      )}
    </div>
  )
}
```
