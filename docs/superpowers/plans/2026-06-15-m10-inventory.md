# M10 — Kho & Định lượng Nguyên liệu — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Quản lý tồn kho nguyên liệu theo công thức (recipe) gắn với product/size. Khi đơn hàng được finalize (PAID), hệ thống tự động trừ tồn kho đúng định lượng. Màn hình `/inventory` cho phép CRUD nguyên liệu, thiết lập công thức theo món, và nhập/điều chỉnh kho.

**Architecture:** Theo M1–M8 — Rust sở hữu SQLite, expose Tauri commands (`commands/inventory.rs` → `repo/inventory.rs` → SQLite, transaction cho mọi thao tác ghi); deduction service được gọi trong `repo/payments.rs::finalize_order` bằng 1 dòng hook sau `orders::mark_paid`; React gọi qua `lib/api/inventory.ts`, màn hình `/inventory` 3 tab dùng shadcn.

**Tech Stack:** rusqlite + rusqlite_migration (sẵn), React + shadcn/ui (sẵn), cargo test in-memory SQLite.

**Quyết định thiết kế:**
- Đơn vị tồn kho = **số nguyên** (đơn vị nhỏ nhất: g, ml, cái). Tránh số thực, tránh lỗi làm tròn.
- Toppings **chưa định lượng** trong M10 (để sau); chỉ định lượng theo product + size_name.
- `recipe_items.size_name IS NULL` = công thức áp dụng mọi size của product đó.
- Khi không có recipe → deduction bỏ qua (không báo lỗi, không block finalize). Cửa hàng chỉ dùng inventory khi đã thiết lập công thức.
- `stock_movements` ghi nhật ký mọi thay đổi tồn kho để truy vết.
- `low_stock_threshold = 0` nghĩa là tắt cảnh báo cho nguyên liệu đó.

**Spec đầy đủ:** `docs/superpowers/specs/2026-06-15-m10-inventory-design.md` (chưa có — plan này là nguồn sự thật).

-- M10: Kho & định lượng nguyên liệu
-- Đơn vị tồn = số nguyên (đơn vị nhỏ nhất: g / ml / cái / v.v.)

CREATE TABLE ingredients (
  id                  INTEGER PRIMARY KEY,
  name                TEXT    NOT NULL,
  unit                TEXT    NOT NULL,          -- 'g' | 'ml' | 'cái' | tuỳ ý
  stock_qty           INTEGER NOT NULL DEFAULT 0,
  low_stock_threshold INTEGER NOT NULL DEFAULT 0, -- 0 = tắt cảnh báo
  is_active           INTEGER NOT NULL DEFAULT 1,
  sort_order          INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE recipe_items (
  id            INTEGER PRIMARY KEY,
  product_id    INTEGER NOT NULL REFERENCES products(id)  ON DELETE CASCADE,
  size_name     TEXT    NULL,                              -- NULL = áp mọi size
  ingredient_id INTEGER NOT NULL REFERENCES ingredients(id) ON DELETE CASCADE,
  qty_per_unit  INTEGER NOT NULL                          -- lượng trừ khi bán 1 đơn vị
);
CREATE INDEX idx_recipe_product ON recipe_items(product_id);
CREATE INDEX idx_recipe_ingredient ON recipe_items(ingredient_id);
-- Mỗi (product, size_name, ingredient) là duy nhất
CREATE UNIQUE INDEX uq_recipe_line
  ON recipe_items(product_id, COALESCE(size_name,''), ingredient_id);

CREATE TABLE stock_movements (
  id            INTEGER PRIMARY KEY,
  ingredient_id INTEGER NOT NULL REFERENCES ingredients(id) ON DELETE CASCADE,
  delta         INTEGER NOT NULL,    -- dương = nhập, âm = xuất
  reason        TEXT    NOT NULL,    -- 'SALE' | 'RESTOCK' | 'ADJUST'
  order_id      INTEGER NULL REFERENCES orders(id) ON DELETE SET NULL,
  created_at    TEXT    NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ','now'))
);
CREATE INDEX idx_movements_ingredient ON stock_movements(ingredient_id);
CREATE INDEX idx_movements_order      ON stock_movements(order_id);
```

fn row_to_ingredient(row: &rusqlite::Row) -> rusqlite::Result<Ingredient> {
    Ok(Ingredient {
        id:                  row.get("id")?,
        name:                row.get("name")?,
        unit:                row.get("unit")?,
        stock_qty:           row.get("stock_qty")?,
        low_stock_threshold: row.get("low_stock_threshold")?,
        is_active:           row.get::<_, i64>("is_active")? != 0,
        sort_order:          row.get("sort_order")?,
    })

fn load_ingredient(conn: &Connection, id: i64) -> rusqlite::Result<Ingredient> {
    conn.query_row("SELECT * FROM ingredients WHERE id=?1", [id], row_to_ingredient)

pub fn create_ingredient(conn: &Connection, input: IngredientInput) -> rusqlite::Result<Ingredient> {
    let name = input.name.trim();
    if name.is_empty() {
        return Err(rusqlite::Error::InvalidParameterName("name rỗng".into()));
    }
    if input.qty_per_unit_check() {}  // placeholder — xoá dòng này, chỉ validate trong recipe

pub fn list_ingredients(conn: &Connection, include_inactive: bool) -> rusqlite::Result<Vec<Ingredient>> {
    let sql = if include_inactive {
        "SELECT * FROM ingredients ORDER BY sort_order, id"
    } else {
        "SELECT * FROM ingredients WHERE is_active=1 ORDER BY sort_order, id"
    };
    let mut stmt = conn.prepare(sql)?;
    stmt.query_map([], row_to_ingredient)?.collect()
}

pub fn update_ingredient(conn: &Connection, id: i64, input: IngredientInput) -> rusqlite::Result<Ingredient> {
    let name = input.name.trim();
    if name.is_empty() {
        return Err(rusqlite::Error::InvalidParameterName("name rỗng".into()));
    }
    conn.execute(
        "UPDATE ingredients SET name=?2, unit=?3, low_stock_threshold=?4, sort_order=?5 WHERE id=?1",
        params![id, name, input.unit.trim(), input.low_stock_threshold, input.sort_order],
    )?;
    load_ingredient(conn, id)
}

pub fn set_ingredient_active(conn: &Connection, id: i64, is_active: bool) -> rusqlite::Result<()> {
    conn.execute(
        "UPDATE ingredients SET is_active=?2 WHERE id=?1",
        params![id, is_active as i64],
    )?;
    Ok(())
}

pub fn delete_ingredient(conn: &Connection, id: i64) -> rusqlite::Result<()> {
    // Kiểm tra còn recipe không
    let n: i64 = conn.query_row(
        "SELECT count(*) FROM recipe_items WHERE ingredient_id=?1",
        [id],
        |r| r.get(0),
    )?;
    if n > 0 {
        return Err(rusqlite::Error::InvalidParameterName(
            "Nguyên liệu đang dùng trong công thức — hãy xoá công thức trước".into(),
        ));
    }
    conn.execute("DELETE FROM ingredients WHERE id=?1", [id])?;
    Ok(())
}

pub fn list_low_stock(conn: &Connection) -> rusqlite::Result<Vec<Ingredient>> {
    // low_stock_threshold > 0 AND stock_qty <= low_stock_threshold
    let mut stmt = conn.prepare(
        "SELECT * FROM ingredients
         WHERE is_active=1 AND low_stock_threshold > 0 AND stock_qty <= low_stock_threshold
         ORDER BY sort_order, id",
    )?;
    stmt.query_map([], row_to_ingredient)?.collect()
}

/// Điều chỉnh tồn kho và ghi nhật ký. delta dương = nhập, âm = xuất.
pub fn adjust_stock(
    conn: &Connection,
    ingredient_id: i64,
    delta: i64,
    reason: &str,
    order_id: Option<i64>,
) -> rusqlite::Result<Ingredient> {
    conn.execute_batch("BEGIN")?;
    let result = (|| {
        conn.execute(
            "UPDATE ingredients SET stock_qty = stock_qty + ?2 WHERE id=?1",
            params![ingredient_id, delta],
        )?;
        conn.execute(
            "INSERT INTO stock_movements(ingredient_id, delta, reason, order_id)
             VALUES (?1, ?2, ?3, ?4)",
            params![ingredient_id, delta, reason, order_id],
        )?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => {
            conn.execute_batch("COMMIT")?;
            load_ingredient(conn, ingredient_id)
        }
        Err(e) => {
            let _ = conn.execute_batch("ROLLBACK");
            Err(e)
        }
    }
}

/// Upsert: nếu (product_id, COALESCE(size_name,''), ingredient_id) đã có → update qty_per_unit.
pub fn upsert_recipe_item(
    conn: &Connection,
    input: RecipeItemInput,
) -> rusqlite::Result<RecipeItem> {
    if input.qty_per_unit <= 0 {
        return Err(rusqlite::Error::InvalidParameterName(
            "qty_per_unit phải > 0".into(),
        ));
    }
    conn.execute(
        "INSERT INTO recipe_items(product_id, size_name, ingredient_id, qty_per_unit)
         VALUES (?1, ?2, ?3, ?4)
         ON CONFLICT(product_id, COALESCE(size_name,''), ingredient_id)
         DO UPDATE SET qty_per_unit = excluded.qty_per_unit",
        params![input.product_id, input.size_name, input.ingredient_id, input.qty_per_unit],
    )?;
    // Lấy id của row vừa upsert
    let id: i64 = conn.query_row(
        "SELECT id FROM recipe_items
         WHERE product_id=?1 AND COALESCE(size_name,'')=COALESCE(?2,'') AND ingredient_id=?3",
        params![input.product_id, input.size_name, input.ingredient_id],
        |r| r.get(0),
    )?;
    conn.query_row(
        "SELECT r.id, r.product_id, r.size_name, r.ingredient_id,
                i.name AS ingredient_name, i.unit, r.qty_per_unit
         FROM recipe_items r
         JOIN ingredients i ON i.id = r.ingredient_id
         WHERE r.id = ?1",
        [id],
        row_to_recipe_item,
    )
}

    conn.execute("DELETE FROM recipe_items WHERE id=?1", [id])?;
    Ok(())

/// Struct nội bộ để gom tổng deduction trước khi ghi DB.
struct Deduction {
    ingredient_id: i64,
    total_qty: i64,
}

/// Trừ tồn kho theo recipe cho một order đã PAID.
/// - Lấy order_items: (product_id, size_name, quantity).
/// - Với mỗi item, lấy recipe_items khớp (product_id AND (size_name=item.size_name OR size_name IS NULL)).
///   Nếu có cả size cụ thể và NULL → ưu tiên size cụ thể (lấy row có size_name = item.size_name).
/// - Gom Σ qty_per_unit × quantity theo ingredient_id.
/// - Ghi stock_movements + update ingredients.stock_qty trong 1 transaction.
/// - Nếu không có recipe nào → trả Ok(()) (noop).
pub fn deduct_stock_for_order(conn: &Connection, order_id: i64) -> rusqlite::Result<()> {
    // 1. Lấy order_items
    let items: Vec<(Option<i64>, Option<String>, i64)> = {
        let mut stmt = conn.prepare(
            "SELECT product_id, size_name, quantity FROM order_items WHERE order_id=?1",
        )?;
        stmt.query_map([order_id], |row| {
            Ok((row.get::<_, Option<i64>>(0)?, row.get::<_, Option<String>>(1)?, row.get::<_, i64>(2)?))
        })?
        .collect::<rusqlite::Result<_>>()?
    };

    // 2. Gom deductions theo ingredient_id
    use std::collections::HashMap;
    let mut deductions: HashMap<i64, i64> = HashMap::new();

    for (product_id_opt, size_name_opt, quantity) in &items {
        let product_id = match product_id_opt {
            Some(id) => *id,
            None => continue, // item không gắn product_id → bỏ qua
        };

        // Lấy recipe: ưu tiên size cụ thể, fallback NULL
        // Query lấy tất cả recipe của (product_id) với size = item.size_name HOẶC NULL
        let recipes: Vec<(i64, Option<String>, i64)> = {
            let mut stmt = conn.prepare(
                "SELECT ingredient_id, size_name, qty_per_unit
                 FROM recipe_items
                 WHERE product_id = ?1
                   AND (size_name = ?2 OR size_name IS NULL)
                 ORDER BY size_name DESC NULLS LAST",
                // DESC: cụ thể (non-null) trước NULL → khi group, lấy cụ thể
            )?;
            stmt.query_map(
                params![product_id, size_name_opt],
                |row| Ok((row.get::<_, i64>(0)?, row.get::<_, Option<String>>(1)?, row.get::<_, i64>(2)?)),
            )?
            .collect::<rusqlite::Result<_>>()?
        };

        // Gom: với mỗi ingredient_id, chọn size cụ thể (non-null) trước NULL
        let mut seen: HashMap<i64, i64> = HashMap::new(); // ingredient_id → qty_per_unit
        for (ing_id, sname, qty) in recipes {
            seen.entry(ing_id).or_insert_with(|| {
                // recipes đã sort: cụ thể trước, nên lấy first occurrence
                let _ = sname; // chỉ dùng để đánh dấu
                qty
            });
        }

        for (ing_id, qty_per_unit) in seen {
            *deductions.entry(ing_id).or_insert(0) += qty_per_unit * quantity;
        }
    }

    if deductions.is_empty() {
        return Ok(());
    }

    // 3. Ghi vào DB trong transaction
    conn.execute_batch("BEGIN")?;
    let result = (|| {
        for (ingredient_id, total_qty) in &deductions {
            conn.execute(
                "UPDATE ingredients SET stock_qty = stock_qty - ?2 WHERE id=?1",
                params![ingredient_id, total_qty],
            )?;
            conn.execute(
                "INSERT INTO stock_movements(ingredient_id, delta, reason, order_id)
                 VALUES (?1, ?2, 'SALE', ?3)",
                params![ingredient_id, -total_qty, order_id],
            )?;
        }
        Ok::<(), rusqlite::Error>(())
    })();

    match result {
        Ok(()) => {
            conn.execute_batch("COMMIT")?;
            Ok(())
        }
        Err(e) => {
            let _ = conn.execute_batch("ROLLBACK");
            Err(e)
        }
    }
}
```

> **Lưu ý SQLite `ORDER BY ... NULLS LAST`:** SQLite 3.30+ hỗ trợ `NULLS LAST`. Nếu cần tương thích rộng hơn, thay bằng `ORDER BY CASE WHEN size_name IS NULL THEN 1 ELSE 0 END ASC` (NULL sau).

                order_id:      row.get("order_id")?,
                created_at:    row.get("created_at")?,
            })
        },
    )

    .collect::<rusqlite::Result<_>>()

}

  type Ingredient, type StockMovement,

type Reason = "RESTOCK" | "ADJUST"

export default function StockTab() {
  const [ingredients, setIngredients] = useState<Ingredient[]>([])
  const [selectedId, setSelectedId]   = useState<number>(0)
  const [delta, setDelta]             = useState<number>(0)
  const [reason, setReason]           = useState<Reason>("RESTOCK")
  const [movements, setMovements]     = useState<StockMovement[]>([])
  const [error, setError]             = useState<string | null>(null)
  const [success, setSuccess]         = useState<string | null>(null)

  async function loadIngredients() {
    try {
      const ings = await listIngredients(false)
      setIngredients(ings)
      if (ings.length > 0 && selectedId === 0) setSelectedId(ings[0].id)
    } catch (e) { setError(String(e)) }
  }

  async function loadMovements(id: number) {
    try { setMovements(await listStockMovements(id, 30)) }
    catch (e) { setError(String(e)) }
  }

  useEffect(() => { loadIngredients() }, [])

  useEffect(() => {
    if (selectedId > 0) loadMovements(selectedId)

      await adjustStock(selectedId, delta, reason)
      setDelta(0)
      setSuccess(`Đã điều chỉnh ${delta > 0 ? "+" : ""}${delta} thành công`)
      await loadIngredients()
      await loadMovements(selectedId)
    } catch (e) { setError(String(e)) }

      <div className="flex flex-col gap-1">

            placeholder="Số lượng (±)"
            value={delta}
            onChange={(e) => setDelta(Number(e.target.value))}
          />
          <select
            className="rounded-md border px-3 py-2 text-sm"
            value={reason}
            onChange={(e) => setReason(e.target.value as Reason)}

        <p className="mt-1 text-xs text-muted-foreground">

        </p>
      </div>

      {error   && <p className="text-sm text-destructive">{error}</p>}
      {success && <p className="text-sm text-green-600">{success}</p>}

      {/* Nhật ký gần nhất */}
      <div>
        <p className="mb-2 text-sm font-medium">Lịch sử 30 lần gần nhất</p>
        <ul className="divide-y rounded-md border text-sm">
          {movements.length === 0 && (
            <li className="p-3 text-muted-foreground">Chưa có nhật ký.</li>
          )}
          {movements.map((m) => (
            <li key={m.id} className="flex items-center justify-between p-3">
              <span className={m.delta >= 0 ? "text-green-600" : "text-destructive"}>
                {m.delta > 0 ? "+" : ""}{m.delta}
              </span>
              <span className="text-muted-foreground">{m.reason}</span>
              <span className="text-muted-foreground">{m.created_at.replace("T", " ").replace("Z", "")}</span>
            </li>
          ))}
