fn row_to_category(row: &rusqlite::Row) -> rusqlite::Result<Category> {
    Ok(Category {
        id: row.get("id")?,
        name: row.get("name")?,
        sort_order: row.get("sort_order")?,
        is_active: row.get::<_, i64>("is_active")? != 0,
    })
}

pub fn create_category(conn: &Connection, name: &str, sort_order: i64) -> rusqlite::Result<Category> {
    let name = name.trim();
    if name.is_empty() {
        return Err(rusqlite::Error::InvalidParameterName("name rỗng".into()));
    }
    conn.execute(
        "INSERT INTO categories(name, sort_order) VALUES (?1, ?2)",
        params![name, sort_order],
    )?;
    let id = conn.last_insert_rowid();
    conn.query_row("SELECT * FROM categories WHERE id=?1", [id], row_to_category)
}

pub fn list_categories(conn: &Connection, include_inactive: bool) -> rusqlite::Result<Vec<Category>> {
    let sql = if include_inactive {
        "SELECT * FROM categories ORDER BY sort_order, id"
    } else {
        "SELECT * FROM categories WHERE is_active=1 ORDER BY sort_order, id"
    };
    let mut stmt = conn.prepare(sql)?;
    let rows = stmt.query_map([], row_to_category)?;
    rows.collect()
}

pub fn update_category(
    conn: &Connection,
    id: i64,
    name: &str,
    sort_order: i64,
    is_active: bool,
) -> rusqlite::Result<()> {
    let name = name.trim();
    if name.is_empty() {
        return Err(rusqlite::Error::InvalidParameterName("name rỗng".into()));
    }
    conn.execute(
        "UPDATE categories SET name=?2, sort_order=?3, is_active=?4 WHERE id=?1",
        params![id, name, sort_order, is_active as i64],
    )?;
    Ok(())
}

fn load_sizes(conn: &Connection, product_id: i64) -> rusqlite::Result<Vec<ProductSize>> {
    let mut stmt = conn.prepare(
        "SELECT id, name, price_delta, is_default FROM product_sizes WHERE product_id=?1 ORDER BY id",
    )?;
    stmt.query_map([product_id], |row| {
        Ok(ProductSize {
            id: row.get("id")?,
            name: row.get("name")?,
            price_delta: row.get("price_delta")?,
            is_default: row.get::<_, i64>("is_default")? != 0,
        })
    })?
    .collect()
}

fn load_product(conn: &Connection, id: i64) -> rusqlite::Result<ProductWithSizes> {
    let mut p = conn.query_row("SELECT * FROM products WHERE id=?1", [id], |row| {
        Ok(ProductWithSizes {
            id: row.get("id")?,
            category_id: row.get("category_id")?,
            name: row.get("name")?,
            base_price: row.get("base_price")?,
            description: row.get("description")?,
            image_path: row.get("image_path")?,
            is_active: row.get::<_, i64>("is_active")? != 0,
            sort_order: row.get("sort_order")?,
            sizes: Vec::new(),
        })
    })?;
    p.sizes = load_sizes(conn, id)?;
    Ok(p)
}

/// Chuẩn hoá: nếu có size mà không có default → đặt size đầu là default; chỉ giữ 1 default.
fn normalize_default(sizes: &mut [SizeInput]) {
    if sizes.is_empty() {
        return;
    }
    let has_default = sizes.iter().any(|s| s.is_default);
    let mut first = true;
    for s in sizes.iter_mut() {
        if !has_default {
            s.is_default = first;
        } else {
            // chỉ giữ default đầu tiên
            if s.is_default && !first {
                // sẽ xử lý ở dưới
            }
        }
        first = false;
    }
    // nếu input có nhiều default, chỉ giữ cái đầu
    let mut seen = false;
    for s in sizes.iter_mut() {
        if s.is_default {
            if seen {
                s.is_default = false;
            }
            seen = true;
        }
    }
}

fn insert_sizes(conn: &Connection, product_id: i64, mut sizes: Vec<SizeInput>) -> rusqlite::Result<()> {
    normalize_default(&mut sizes);
    for s in sizes {
        if s.name.trim().is_empty() || s.price_delta < 0 {
            return Err(rusqlite::Error::InvalidParameterName("size không hợp lệ".into()));
        }
        conn.execute(
            "INSERT INTO product_sizes(product_id, name, price_delta, is_default) VALUES (?1,?2,?3,?4)",
            params![product_id, s.name.trim(), s.price_delta, s.is_default as i64],
        )?;
    }
    Ok(())
}

fn validate_product(conn: &Connection, input: &ProductInput) -> rusqlite::Result<()> {
    if input.name.trim().is_empty() || input.base_price < 0 {
        return Err(rusqlite::Error::InvalidParameterName("name/base_price không hợp lệ".into()));
    }
    let cat_exists: i64 = conn.query_row(
        "SELECT count(*) FROM categories WHERE id=?1",
        [input.category_id],
        |r| r.get(0),
    )?;
    if cat_exists == 0 {
        return Err(rusqlite::Error::InvalidParameterName("category_id không tồn tại".into()));
    }
    Ok(())
}

pub fn create_product(conn: &Connection, input: ProductInput) -> rusqlite::Result<ProductWithSizes> {
    validate_product(conn, &input)?;
    conn.execute_batch("BEGIN")?;
    let res = (|| {
        conn.execute(
            "INSERT INTO products(category_id, name, base_price, description, image_path, sort_order)
             VALUES (?1,?2,?3,?4,?5,?6)",
            params![input.category_id, input.name.trim(), input.base_price, input.description, input.image_path, input.sort_order],
        )?;
        let id = conn.last_insert_rowid();
        insert_sizes(conn, id, input.sizes)?;
        Ok::<i64, rusqlite::Error>(id)
    })();
    match res {
        Ok(id) => { conn.execute_batch("COMMIT")?; load_product(conn, id) }
        Err(e) => { let _ = conn.execute_batch("ROLLBACK"); Err(e) }
    }
}

pub fn update_product(conn: &Connection, id: i64, input: ProductInput) -> rusqlite::Result<ProductWithSizes> {
    validate_product(conn, &input)?;
    conn.execute_batch("BEGIN")?;
    let res = (|| {
        conn.execute(
            "UPDATE products SET category_id=?2, name=?3, base_price=?4, description=?5, image_path=?6, sort_order=?7 WHERE id=?1",
            params![id, input.category_id, input.name.trim(), input.base_price, input.description, input.image_path, input.sort_order],
        )?;
        conn.execute("DELETE FROM product_sizes WHERE product_id=?1", [id])?;
        insert_sizes(conn, id, input.sizes)?;
        Ok::<(), rusqlite::Error>(())
    })();
    match res {
        Ok(()) => { conn.execute_batch("COMMIT")?; load_product(conn, id) }
        Err(e) => { let _ = conn.execute_batch("ROLLBACK"); Err(e) }
    }
}

pub fn set_product_active(conn: &Connection, id: i64, is_active: bool) -> rusqlite::Result<()> {

}

pub fn delete_product(conn: &Connection, id: i64) -> rusqlite::Result<()> {
    conn.execute("DELETE FROM products WHERE id=?1", [id])?; // sizes CASCADE
    Ok(())
}

pub fn list_products(conn: &Connection, category_id: Option<i64>, include_inactive: bool) -> rusqlite::Result<Vec<ProductWithSizes>> {
    let mut sql = String::from("SELECT id FROM products WHERE 1=1");
    if !include_inactive { sql.push_str(" AND is_active=1"); }
    if category_id.is_some() { sql.push_str(" AND category_id=?1"); }
    sql.push_str(" ORDER BY sort_order, id");
    let mut stmt = conn.prepare(&sql)?;
    let ids: Vec<i64> = if let Some(cid) = category_id {
        stmt.query_map([cid], |r| r.get(0))?.collect::<rusqlite::Result<_>>()?
    } else {
        stmt.query_map([], |r| r.get(0))?.collect::<rusqlite::Result<_>>()?
    };
    ids.into_iter().map(|id| load_product(conn, id)).collect()
}

pub fn delete_category(conn: &Connection, id: i64) -> rusqlite::Result<()> {
    let n: i64 = conn.query_row("SELECT count(*) FROM products WHERE category_id=?1", [id], |r| r.get(0))?;
    if n > 0 {
        return Err(rusqlite::Error::InvalidParameterName("Danh mục còn món — hãy vô hiệu hoá thay vì xoá".into()));
    }
    conn.execute("DELETE FROM categories WHERE id=?1", [id])?;
    Ok(())
}

use base64::Engine;
use std::io::Write;
use tauri::Manager;

fn images_dir(app: &tauri::AppHandle) -> Result<std::path::PathBuf, String> {
    let dir = app.path().app_data_dir().map_err(|e| e.to_string())?.join("images");
    std::fs::create_dir_all(&dir).map_err(|e| e.to_string())?;
    Ok(dir)
}

/// Copy file ảnh nguồn vào app_data_dir/images, trả về đường dẫn tương đối "images/<uuid>.<ext>".
#[tauri::command]
pub fn save_product_image(app: tauri::AppHandle, source_path: String) -> Result<String, String> {
    let src = std::path::Path::new(&source_path);
    let ext = src.extension().and_then(|e| e.to_str()).unwrap_or("png").to_lowercase();
    let bytes = std::fs::read(src).map_err(|e| e.to_string())?;
    let name = format!("{}.{}", uuid::Uuid::new_v4(), ext);
    let dir = images_dir(&app)?;
    let mut f = std::fs::File::create(dir.join(&name)).map_err(|e| e.to_string())?;
    f.write_all(&bytes).map_err(|e| e.to_string())?;
    Ok(format!("images/{name}"))
}

/// Đọc ảnh theo đường dẫn tương đối, trả về data-URL để hiển thị trong webview.
#[tauri::command]
pub fn read_image_data_url(app: tauri::AppHandle, relative_path: String) -> Result<String, String> {
    let full = app.path().app_data_dir().map_err(|e| e.to_string())?.join(&relative_path);
    let bytes = std::fs::read(&full).map_err(|e| e.to_string())?;
    let ext = std::path::Path::new(&relative_path).extension().and_then(|e| e.to_str()).unwrap_or("png");
    let mime = match ext.to_lowercase().as_str() {
        "jpg" | "jpeg" => "image/jpeg",
        "webp" => "image/webp",
        _ => "image/png",
    };
    let b64 = base64::engine::general_purpose::STANDARD.encode(&bytes);
    Ok(format!("data:{mime};base64,{b64}"))
}

export default function ProductForm({ categories, editing, onDone }: Props) {
  const [name, setName] = useState(editing?.name ?? "")
  const [categoryId, setCategoryId] = useState(editing?.category_id ?? categories[0]?.id ?? 0)
  const [basePrice, setBasePrice] = useState(editing?.base_price ?? 0)
  const [description, setDescription] = useState(editing?.description ?? "")
  const [imagePath, setImagePath] = useState<string | null>(editing?.image_path ?? null)

      const selected = await open({ multiple: false, filters: [{ name: "Ảnh", extensions: ["png", "jpg", "jpeg", "webp"] }] })

    } catch (e) { setError(String(e)) }

      <Input placeholder="Tên món" value={name} onChange={(e) => setName(e.target.value)} />

      </select>
      <Input type="number" placeholder="Giá gốc (VND)" value={basePrice} onChange={(e) => setBasePrice(Number(e.target.value))} />
      <Input placeholder="Mô tả" value={description} onChange={(e) => setDescription(e.target.value)} />
      <div className="flex items-center gap-3">
        <Button type="button" variant="outline" onClick={pickImage}>Chọn ảnh</Button>
        {preview && <img src={preview} alt="" className="h-16 w-16 rounded object-cover" />}
      </div>
      <div className="flex flex-col gap-2">
        <div className="flex items-center justify-between">
          <span className="text-sm font-medium">Size</span>
          <Button type="button" variant="ghost" size="sm" onClick={() => setSizes((p) => [...p, { name: "", price_delta: 0, is_default: p.length === 0 }])}>+ Size</Button>
        </div>
        {sizes.map((s, i) => (
          <div key={i} className="flex items-center gap-2">
            <Input placeholder="Tên size" value={s.name} onChange={(e) => setSizes((p) => p.map((x, j) => j === i ? { ...x, name: e.target.value } : x))} />
            <Input type="number" placeholder="Phụ giá" value={s.price_delta} onChange={(e) => setSizes((p) => p.map((x, j) => j === i ? { ...x, price_delta: Number(e.target.value) } : x))} />
            <label className="flex items-center gap-1 text-sm"><input type="radio" name="defsize" checked={s.is_default} onChange={() => setDefault(i)} /> Mặc định</label>
            <Button type="button" variant="ghost" size="sm" onClick={() => setSizes((p) => p.filter((_, j) => j !== i))}>x</Button>
          </div>
        ))}
      </div>
      {error && <p className="text-sm text-destructive">{error}</p>}
      <div className="flex justify-end gap-2">
        <Button variant="outline" onClick={onDone}>Huỷ</Button>
        <Button onClick={onSave}>Lưu</Button>
      </div>
    </div>
  )
}

Hiển thị thumbnail bằng `readImageDataUrl` khi `image_path` có. Tất cả async bọc try/catch.

```tsx
import { useEffect, useState } from "react"
import { Dialog, DialogContent, DialogTrigger, DialogTitle } from "@/components/ui/dialog"
import { Button } from "@/components/ui/button"
import { Switch } from "@/components/ui/switch"
import ProductForm from "@/routes/menu/ProductForm"
import { formatVnd } from "@/lib/format"
import {
  listCategories, listProducts, setProductActive, deleteProduct, readImageDataUrl,
  type Category, type Product,
} from "@/lib/api/menu"

export default function ProductTab() {
  const [cats, setCats] = useState<Category[]>([])
  const [items, setItems] = useState<Product[]>([])
  const [thumbs, setThumbs] = useState<Record<number, string>>({})
  const [open, setOpen] = useState(false)
  const [editing, setEditing] = useState<Product | null>(null)
  const [error, setError] = useState<string | null>(null)

  async function refresh() {
    try {
      setCats(await listCategories(true))
      const ps = await listProducts(null, true)
      setItems(ps)
      const t: Record<number, string> = {}
      for (const p of ps) if (p.image_path) t[p.id] = await readImageDataUrl(p.image_path)
      setThumbs(t)
    } catch (e) { setError(String(e)) }
  }
  useEffect(() => { refresh() }, [])

  return (
    <div className="flex flex-col gap-3 py-4">
      <div className="flex justify-end">
        <Dialog open={open} onOpenChange={setOpen}>
          <DialogTrigger asChild><Button onClick={() => setEditing(null)}>Thêm món</Button></DialogTrigger>
          <DialogContent>
            <DialogTitle>{editing ? "Sửa món" : "Thêm món"}</DialogTitle>
            <ProductForm categories={cats} editing={editing} onDone={() => { setOpen(false); refresh() }} />
          </DialogContent>
        </Dialog>
      </div>
      {error && <p className="text-sm text-destructive">{error}</p>}
      <ul className="divide-y rounded-md border">
        {items.map((p) => (
          <li key={p.id} className="flex items-center justify-between p-3">
            <div className="flex items-center gap-3">
              {thumbs[p.id]
                ? <img src={thumbs[p.id]} alt="" className="h-12 w-12 rounded object-cover" />
                : <div className="h-12 w-12 rounded bg-muted" />}
              <div>
                <div className={p.is_active ? "font-medium" : "font-medium text-muted-foreground line-through"}>{p.name}</div>
                <div className="text-sm text-muted-foreground">{formatVnd(p.base_price)} · {p.sizes.length} size</div>
              </div>
            </div>
            <div className="flex items-center gap-3">
              <Switch checked={p.is_active} onCheckedChange={(v) => setProductActive(p.id, v).then(refresh)} />
              <Button variant="ghost" size="sm" onClick={() => { setEditing(p); setOpen(true) }}>Sửa</Button>
              <Button variant="ghost" size="sm" onClick={() => deleteProduct(p.id).then(refresh)}>Xoá</Button>
            </div>
          </li>
        ))}
      </ul>
    </div>
  )
}
```

```tsx
import { useEffect, useState } from "react"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Switch } from "@/components/ui/switch"
import { formatVnd } from "@/lib/format"
import {
  listToppings, createTopping, updateTopping, deleteTopping, type Topping,
} from "@/lib/api/menu"

export default function ToppingTab() {
  const [items, setItems] = useState<Topping[]>([])
  const [name, setName] = useState("")
  const [price, setPrice] = useState(0)
  const [error, setError] = useState<string | null>(null)

  async function refresh() {
    try { setItems(await listToppings(true)) } catch (e) { setError(String(e)) }
  }
  useEffect(() => { refresh() }, [])

  async function onAdd() {
    setError(null)
    if (!name.trim()) return
    try { await createTopping(name.trim(), price, items.length); setName(""); setPrice(0); await refresh() }
    catch (e) { setError(String(e)) }
  }

  return (
    <div className="flex flex-col gap-3 py-4">
      <div className="flex gap-2">
        <Input placeholder="Tên topping" value={name} onChange={(e) => setName(e.target.value)} />
        <Input type="number" placeholder="Giá" value={price} onChange={(e) => setPrice(Number(e.target.value))} />
        <Button onClick={onAdd}>Thêm</Button>
      </div>
      {error && <p className="text-sm text-destructive">{error}</p>}
      <ul className="divide-y rounded-md border">
        {items.map((t) => (
          <li key={t.id} className="flex items-center justify-between p-3">
            <span className={t.is_active ? "" : "text-muted-foreground line-through"}>{t.name} — {formatVnd(t.price)}</span>
            <div className="flex items-center gap-3">
              <Switch checked={t.is_active} onCheckedChange={(v) => updateTopping(t.id, t.name, t.price, t.sort_order, v).then(refresh)} />
              <Button variant="ghost" size="sm" onClick={() => deleteTopping(t.id).then(refresh)}>Xoá</Button>
            </div>
          </li>
        ))}
      </ul>
    </div>
  )
}
```
