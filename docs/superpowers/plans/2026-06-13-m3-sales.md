// services/order.rs
use crate::domain::orders::{OrderItem, OrderItemTopping, OrderItemInput, OrderItemToppingInput};

/// Tính line_total cho một item theo contract §4:
/// line_total = (unit_price + Σ(topping.price × topping.quantity)) × quantity − line_discount
/// Clamp: max(0, line_total)
pub fn compute_line_total(item: &OrderItemInput) -> i64 {
    let toppings_cost: i64 = item
        .toppings
        .iter()
        .map(|t| t.price * t.quantity)
        .sum();
    let raw = (item.unit_price + toppings_cost) * item.quantity - item.line_discount;
    raw.max(0)
}

/// Tính subtotal = Σ line_total của tất cả items (đã tính sẵn line_total truyền vào).
pub fn compute_subtotal(line_totals: &[i64]) -> i64 {
    line_totals.iter().sum()
}

/// Tính total = max(0, subtotal − discount_total).
/// M3 không có discount_total (M4 xử lý); truyền 0 là đúng.
pub fn compute_total(subtotal: i64, discount_total: i64) -> i64 {
    (subtotal - discount_total).max(0)
}

/// Làm tròn về đơn vị rounding_unit (mặc định 1 = không đổi).
/// Ví dụ rounding_unit=500: 34700 → 34500, 34250 → 34000 (floor).
pub fn round_vnd(amount: i64, rounding_unit: i64) -> i64 {
    if rounding_unit <= 1 {
        return amount;
    }
    (amount / rounding_unit) * rounding_unit
}

/// Sinh mã đơn dạng "ORD-YYYYMMDD-HHMMSSmmm" — đủ unique cho single-station.
/// Nếu thêm dep chrono chưa có: dùng std::time::SystemTime thay thế (xem bên dưới).
pub fn gen_order_code() -> String {
    // Dùng std thay vì dep chrono để không thêm crate mới:
    use std::time::{SystemTime, UNIX_EPOCH};
    let ms = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_millis();
    // format: ORD-<unix_ms_hex> → đảm bảo unique + tăng dần
    format!("ORD-{ms:X}")
}

/// Lấy timestamp UTC hiện tại dạng ISO-8601 để lưu vào DB.
pub fn now_utc() -> String {
    use std::time::{SystemTime, UNIX_EPOCH};
    // format thủ công: YYYY-MM-DDTHH:MM:SSZ
    // Nếu repo có chrono thì dùng: chrono::Utc::now().to_rfc3339()
    // Fallback std:
    let secs = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs();
    // Hàm đơn giản — đủ dùng, không cần chrono:
    // Tính năm/tháng/ngày/giờ/phút/giây từ unix seconds
    format_unix_utc(secs)
}

fn format_unix_utc(secs: u64) -> String {
    // Tính offset từ epoch
    let s = secs % 60;
    let m = (secs / 60) % 60;
    let h = (secs / 3600) % 24;
    let days = secs / 86400;
    // Tính năm/tháng/ngày bằng thuật toán Gregorian đơn giản
    let (year, month, day) = days_to_ymd(days);
    format!("{year:04}-{month:02}-{day:02}T{h:02}:{m:02}:{s:02}Z")
}

fn days_to_ymd(days: u64) -> (u64, u64, u64) {
    // Thuật toán Richards (Civil Date)
    let z = days + 719468;
    let era = z / 146097;
    let doe = z % 146097;
    let yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
    let y = yoe + era * 400;
    let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
    let mp = (5 * doy + 2) / 153;
    let d = doy - (153 * mp + 2) / 5 + 1;
    let m = if mp < 10 { mp + 3 } else { mp - 9 };
    let y = if m <= 2 { y + 1 } else { y };
    (y, m, d)
}

/// Recompute subtotal/discount_total/total từ items hiện tại rồi cập nhật vào orders row.
/// discount_total luôn = 0 trong M3 (M4 xử lý); total = max(0, subtotal - 0).
fn recompute_and_save_totals(conn: &Connection, order_id: i64) -> rusqlite::Result<()> {
    let line_totals: Vec<i64> = {
        let mut stmt = conn.prepare(
            "SELECT line_total FROM order_items WHERE order_id=?1",
        )?;
        stmt.query_map([order_id], |r| r.get(0))?.collect::<rusqlite::Result<_>>()?
    };
    let subtotal = svc::compute_subtotal(&line_totals);
    let total = svc::compute_total(subtotal, 0); // discount_total = 0 in M3
    conn.execute(
        "UPDATE orders SET subtotal=?2, total=?3 WHERE id=?1",
        params![order_id, subtotal, total],
    )?;
    Ok(())
}

/// Thêm một item vào đơn OPEN, recompute totals, trả về Order đầy đủ.
/// Toàn bộ chạy trong một transaction.
pub fn add_order_item(
    conn: &Connection,
    order_id: i64,
    input: OrderItemInput,
) -> rusqlite::Result<Order> {
    // Validate đơn tồn tại và OPEN
    let status: String = conn.query_row(
        "SELECT status FROM orders WHERE id=?1", [order_id], |r| r.get(0),
    )?;
    if status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName(
            "Chỉ thêm item vào đơn OPEN".into(),
        ));
    }

    let line_total = svc::compute_line_total(&input);

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        let item_id = insert_order_item(conn, order_id, &input, line_total)?;
        insert_item_toppings(conn, item_id, &input.toppings)?;
        recompute_and_save_totals(conn, order_id)?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => { conn.execute_batch("COMMIT")?; load_order(conn, order_id) }
        Err(e) => { let _ = conn.execute_batch("ROLLBACK"); Err(e) }
    }
}

/// Cập nhật item (qty/options/note). Xóa và tái tạo toppings, recompute totals.
pub fn update_order_item(
    conn: &Connection,
    item_id: i64,
    input: OrderItemInput,
) -> rusqlite::Result<Order> {
    // Tìm order_id từ item
    let order_id: i64 = conn.query_row(
        "SELECT order_id FROM order_items WHERE id=?1", [item_id], |r| r.get(0),
    )?;
    // Validate đơn OPEN
    let status: String = conn.query_row(
        "SELECT status FROM orders WHERE id=?1", [order_id], |r| r.get(0),
    )?;
    if status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName(
            "Chỉ sửa item trong đơn OPEN".into(),
        ));
    }
    let line_total = svc::compute_line_total(&input);

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        conn.execute(
            "UPDATE order_items SET
                 product_id=?2, product_name=?3, size_name=?4, unit_price=?5,
                 quantity=?6, sugar_level=?7, ice_level=?8, line_note=?9,
                 line_discount=?10, line_total=?11
             WHERE id=?1",
            params![
                item_id, input.product_id, input.product_name, input.size_name,
                input.unit_price, input.quantity, input.sugar_level, input.ice_level,
                input.line_note, input.line_discount, line_total
            ],
        )?;
        // Xóa toppings cũ (CASCADE đã xử lý nếu order_item bị xóa, nhưng ở đây ta update)
        conn.execute("DELETE FROM order_item_toppings WHERE order_item_id=?1", [item_id])?;
        insert_item_toppings(conn, item_id, &input.toppings)?;
        recompute_and_save_totals(conn, order_id)?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => { conn.execute_batch("COMMIT")?; load_order(conn, order_id) }
        Err(e) => { let _ = conn.execute_batch("ROLLBACK"); Err(e) }
    }
}

/// Xóa item (toppings CASCADE). Recompute totals.
pub fn remove_order_item(conn: &Connection, item_id: i64) -> rusqlite::Result<Order> {
    let order_id: i64 = conn.query_row(
        "SELECT order_id FROM order_items WHERE id=?1", [item_id], |r| r.get(0),
    )?;
    let status: String = conn.query_row(
        "SELECT status FROM orders WHERE id=?1", [order_id], |r| r.get(0),
    )?;
    if status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName(
            "Chỉ xóa item trong đơn OPEN".into(),
        ));
    }

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        conn.execute("DELETE FROM order_items WHERE id=?1", [item_id])?;
        recompute_and_save_totals(conn, order_id)?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => { conn.execute_batch("COMMIT")?; load_order(conn, order_id) }
        Err(e) => { let _ = conn.execute_batch("ROLLBACK"); Err(e) }
    }
}

/// Hủy đơn. Đơn phải OPEN. Bàn được nhả (status suy ra từ không còn OPEN order).
pub fn cancel_order(conn: &Connection, id: i64) -> rusqlite::Result<Order> {
    let status: String = conn.query_row(
        "SELECT status FROM orders WHERE id=?1", [id], |r| r.get(0),
    )?;
    if status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName(
            "Chỉ cancel đơn OPEN".into(),
        ));
    }
    conn.execute(
        "UPDATE orders SET status='CANCELLED' WHERE id=?1",
        [id],
    )?;
    load_order(conn, id)
}

/// Chuyển bàn: đổi table_id của đơn OPEN sang bàn trống (không có OPEN order nào).
pub fn transfer_table(conn: &Connection, order_id: i64, to_table_id: i64) -> rusqlite::Result<Order> {
    // Validate đơn OPEN
    let status: String = conn.query_row(
        "SELECT status FROM orders WHERE id=?1", [order_id], |r| r.get(0),
    )?;
    if status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName("Đơn phải OPEN".into()));
    }
    // Kiểm tra bàn đích trống
    let occupied: i64 = conn.query_row(
        "SELECT count(*) FROM orders WHERE table_id=?1 AND status='OPEN'",
        [to_table_id],
        |r| r.get(0),
    )?;
    if occupied > 0 {
        return Err(rusqlite::Error::InvalidParameterName(
            "Bàn đích đang có đơn OPEN".into(),
        ));
    }
    conn.execute(
        "UPDATE orders SET table_id=?2 WHERE id=?1",
        params![order_id, to_table_id],
    )?;
    load_order(conn, order_id)
}

/// Gộp bàn: dồn items từ source_order_ids vào target_order_id, cancel sources.
/// Tất cả trong một transaction. Snapshot names/prices được giữ nguyên (đã trong order_items).
pub fn merge_tables(
    conn: &Connection,
    source_order_ids: Vec<i64>,
    target_order_id: i64,
) -> rusqlite::Result<Order> {
    // Validate target OPEN
    let status: String = conn.query_row(
        "SELECT status FROM orders WHERE id=?1", [target_order_id], |r| r.get(0),
    )?;
    if status != "OPEN" {
        return Err(rusqlite::Error::InvalidParameterName("Đơn đích phải OPEN".into()));
    }
    for &src_id in &source_order_ids {
        let s: String = conn.query_row(
            "SELECT status FROM orders WHERE id=?1", [src_id], |r| r.get(0),
        )?;
        if s != "OPEN" {
            return Err(rusqlite::Error::InvalidParameterName(
                "Đơn nguồn phải OPEN".into(),
            ));
        }
    }

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        for src_id in &source_order_ids {
            // Di chuyển items: update order_id + recompute line_total giữ nguyên (snapshot OK)
            conn.execute(
                "UPDATE order_items SET order_id=?2 WHERE order_id=?1",
                params![src_id, target_order_id],
            )?;
            // Cancel đơn nguồn
            conn.execute(
                "UPDATE orders SET status='CANCELLED' WHERE id=?1",
                [src_id],
            )?;
        }
        recompute_and_save_totals(conn, target_order_id)?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => { conn.execute_batch("COMMIT")?; load_order(conn, target_order_id) }
        Err(e) => { let _ = conn.execute_batch("ROLLBACK"); Err(e) }
    }
}

import { create } from "zustand"
import type { OrderItemInput, OrderItemToppingInput } from "@/lib/api/orders"

export type CartTopping = {
  topping_id: number | null
  topping_name: string
  price: number
  quantity: number
}

export type CartItem = {
  // Draft item trước khi gửi lên Rust
  key: string        // random key để nhận diện trên UI (không phải DB id)
  product_id: number | null
  product_name: string
  size_name: string | null
  unit_price: number    // base_price + size price_delta
  quantity: number
  sugar_level: string | null
  ice_level: string | null
  line_note: string | null
  line_discount: number
  toppings: CartTopping[]
}

type CartState = {
  items: CartItem[]
  addItem: (item: Omit<CartItem, "key">) => void
  updateItem: (key: string, updates: Partial<Omit<CartItem, "key">>) => void
  removeItem: (key: string) => void
  clearCart: () => void
  toOrderItemInput: (item: CartItem) => OrderItemInput
}

function makeKey() {
  return Math.random().toString(36).slice(2) + Date.now().toString(36)
}

export const useCartStore = create<CartState>((set, get) => ({
  items: [],

  addItem: (item) =>
    set((s) => ({ items: [...s.items, { ...item, key: makeKey() }] })),

  updateItem: (key, updates) =>
    set((s) => ({
      items: s.items.map((i) => (i.key === key ? { ...i, ...updates } : i)),
    })),

  removeItem: (key) =>
    set((s) => ({ items: s.items.filter((i) => i.key !== key) })),

  clearCart: () => set({ items: [] }),

  toOrderItemInput: (item): OrderItemInput => ({
    product_id: item.product_id,
    product_name: item.product_name,
    size_name: item.size_name,
    unit_price: item.unit_price,
    quantity: item.quantity,
    sugar_level: item.sugar_level,
    ice_level: item.ice_level,
    line_note: item.line_note,
    line_discount: item.line_discount,
    toppings: item.toppings.map((t) => ({
      topping_id: t.topping_id,
      topping_name: t.topping_name,
      price: t.price,
      quantity: t.quantity,
    })),
  }),
}))

// Helper: tính line_total hiển thị trong cart UI (mirror logic Rust services/order.rs)
export function calcCartItemTotal(item: CartItem): number {
  const toppingsCost = item.toppings.reduce((sum, t) => sum + t.price * t.quantity, 0)
  return Math.max(0, (item.unit_price + toppingsCost) * item.quantity - item.line_discount)
}

export function calcCartSubtotal(items: CartItem[]): number {
  return items.reduce((sum, i) => sum + calcCartItemTotal(i), 0)
}

      ice_level: ice,

      line_discount: 0,
      toppings: selectedToppings,

    onClose()

                >
                  {lvl}
                </button>

          </div>

          <div>
            <p className="text-sm font-medium mb-2">Mức đá</p>

                  key={lvl}
                  onClick={() => setIce(lvl)}

            <p className="text-sm font-medium mb-2">Ghi chú</p>

import { useState } from "react"
import { Button } from "@/components/ui/button"
import { ScrollArea } from "@/components/ui/scroll-area"
import { Separator } from "@/components/ui/separator"
import { formatVnd } from "@/lib/format"
import { useCartStore, calcCartItemTotal, calcCartSubtotal } from "@/store/cart"
import { createOrder, addOrderItem, cancelOrder, type Order } from "@/lib/api/orders"
import OrderTypeSelector from "@/routes/sales/OrderTypeSelector"
import type { OrderType } from "@/lib/api/orders"

type Props = {
  // Danh sách bàn trống để chọn DINE_IN (load từ parent)
  freeTables: { id: number; name: string }[]
  // Callback khi đơn được tạo thành công
  onOrderCreated: (order: Order) => void
}

export default function CartPanel({ freeTables, onOrderCreated }: Props) {
  const { items, removeItem, updateItem, clearCart, toOrderItemInput } = useCartStore()
  const [orderType, setOrderType] = useState<OrderType>("TAKEAWAY")
  const [tableId, setTableId] = useState<number | null>(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const subtotal = calcCartSubtotal(items)

  async function handleCreateOrder() {
    setError(null)
    if (orderType === "DINE_IN" && !tableId) {
      setError("Vui lòng chọn bàn")
      return
    }
    if (items.length === 0) {
      setError("Giỏ hàng trống")
      return
    }
    setLoading(true)
    try {
      // 1. Tạo đơn
      let order = await createOrder({ order_type: orderType, table_id: tableId, note: null })
      // 2. Add từng item
      for (const cartItem of items) {
        order = await addOrderItem(order.id, toOrderItemInput(cartItem))
      }
      clearCart()
      onOrderCreated(order)
    } catch (e) {
      setError(String(e))
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="flex flex-col h-full">
      <div className="flex items-center justify-between px-4 py-3 border-b shrink-0">
        <h2 className="font-semibold">Giỏ hàng</h2>

          </Button>
        )}
      </div>

      <OrderTypeSelector
        value={orderType}
        tableId={tableId}
        tables={freeTables}
        onChange={(t, tid) => { setOrderType(t); setTableId(tid) }}
      />

      <ScrollArea className="flex-1 px-3 py-2">
        {items.length === 0 ? (
          <p className="py-8 text-center text-sm text-muted-foreground">Chưa có món nào</p>
        ) : (
          <ul className="flex flex-col gap-2">
            {items.map((item) => (
              <li key={item.key} className="rounded-lg border p-3">
                <div className="flex items-start justify-between gap-2">
                  <div className="flex-1 min-w-0">
                    <p className="font-medium text-sm leading-snug truncate">{item.product_name}</p>
                    {item.size_name && (
                      <p className="text-xs text-muted-foreground">{item.size_name}</p>
                    )}
                    {(item.sugar_level || item.ice_level) && (
                      <p className="text-xs text-muted-foreground">
                        {[item.sugar_level, item.ice_level].filter(Boolean).join(" · ")}
                      </p>
                    )}
                    {item.toppings.length > 0 && (
                      <p className="text-xs text-muted-foreground">
                        + {item.toppings.map((t) => t.topping_name).join(", ")}
                      </p>
                    )}
                    {item.line_note && (
                      <p className="text-xs italic text-muted-foreground">{item.line_note}</p>
                    )}
                  </div>
                  <div className="flex flex-col items-end gap-1 shrink-0">
                    <span className="text-sm font-semibold">{formatVnd(calcCartItemTotal(item))}</span>
                    <div className="flex items-center gap-1">
                      <button
                        onClick={() => {
                          if (item.quantity === 1) removeItem(item.key)
                          else updateItem(item.key, { quantity: item.quantity - 1 })
                        }}
                        className="h-8 w-8 rounded border text-sm font-bold hover:bg-muted transition-colors"
                      >
                        −
                      </button>
                      <span className="w-6 text-center text-sm">{item.quantity}</span>
                      <button
                        onClick={() => updateItem(item.key, { quantity: item.quantity + 1 })}
                        className="h-8 w-8 rounded border text-sm font-bold hover:bg-muted transition-colors"
                      >
                        +
                      </button>
                    </div>
                  </div>
                </div>
              </li>
            ))}
          </ul>
        )}
      </ScrollArea>

      <Separator />

      <div className="p-4 flex flex-col gap-3 shrink-0">
        <div className="flex justify-between items-center">
          <span className="text-sm text-muted-foreground">Tạm tính</span>
          <span className="font-semibold text-lg">{formatVnd(subtotal)}</span>

          size="lg"
          className="w-full min-h-[52px] text-base"
          disabled={loading || items.length === 0}
          onClick={handleCreateOrder}
        >
          {loading ? "Đang tạo..." : "Tạo đơn"}
        </Button>
      </div>
    </div>
  )
}

// Thay toàn bộ query bằng:
let sql = "
    SELECT t.id, t.area_id, t.name, t.seats, t.sort_order, t.is_active,
           o.id AS open_order_id
    FROM tables t
    LEFT JOIN orders o ON o.table_id = t.id AND o.status = 'OPEN'
    WHERE t.is_active = 1
    ORDER BY t.sort_order, t.id
";
let mut stmt = conn.prepare(sql)?;
stmt.query_map([], |row| {
    let open_order_id: Option<i64> = row.get("open_order_id")?;
    let status = if open_order_id.is_some() {
        TableState::DangPhucVu
    } else {
        TableState::Trong
    };
    let table = Table {
        id: row.get("id")?,
        area_id: row.get("area_id")?,
        name: row.get("name")?,
        seats: row.get("seats")?,
        sort_order: row.get("sort_order")?,
        is_active: row.get::<_, i64>("is_active")? != 0,
    };
    Ok(TableStatus { table, status, open_order_id })
})?
.collect()
