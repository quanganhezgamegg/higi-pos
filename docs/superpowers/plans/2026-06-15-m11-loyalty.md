-- Bảng khách hàng
CREATE TABLE customers (
  id          INTEGER PRIMARY KEY,
  name        TEXT NOT NULL,
  phone       TEXT NOT NULL UNIQUE,
  points      INTEGER NOT NULL DEFAULT 0,
  total_spent INTEGER NOT NULL DEFAULT 0,
  created_at  TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ','now')),
  is_active   INTEGER NOT NULL DEFAULT 1
);
CREATE INDEX idx_customers_phone ON customers(phone);

-- Lịch sử giao dịch điểm
-- kind: 'ACCRUE' (tích điểm khi thanh toán) | 'REDEEM' (đổi điểm giảm giá) | 'ADJUST' (admin điều chỉnh thủ công)
-- points: dương khi cộng, âm khi trừ
CREATE TABLE point_transactions (
  id          INTEGER PRIMARY KEY,
  customer_id INTEGER NOT NULL REFERENCES customers(id),
  order_id    INTEGER,                   -- NULL cho ADJUST không gắn đơn
  kind        TEXT NOT NULL,             -- 'ACCRUE' | 'REDEEM' | 'ADJUST'
  points      INTEGER NOT NULL,          -- +/-, integer VND-based
  created_at  TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ','now')),
  note        TEXT
);
CREATE INDEX idx_pt_customer ON point_transactions(customer_id);
CREATE INDEX idx_pt_order    ON point_transactions(order_id);

-- Gắn khách hàng vào đơn (nullable, không hard FK vì SQLite ALTER ADD COLUMN không hỗ trợ REFERENCES)
-- Ứng dụng tự enforce: customer_id phải tồn tại trong customers khi set != NULL
ALTER TABLE orders ADD COLUMN customer_id INTEGER;
```

> **Lý do không có hard FK trên ALTER ADD COLUMN:** SQLite không cho phép thêm constraint FK vào cột qua ALTER TABLE. Integrity được enforce ở tầng repo Rust (kiểm tra existence trước khi INSERT/UPDATE).

/// Đọc cấu hình loyalty từ settings. Trả về (enabled, accrual_vnd_per_point, redeem_point_value).
/// Mặc định: disabled, 10000 vnd/điểm, 1000 vnd/điểm đổi.
pub fn get_loyalty_config(conn: &Connection) -> rusqlite::Result<(bool, i64, i64)> {
    let enabled = get_setting(conn, "loyalty_enabled")?.unwrap_or_else(|| "0".into()) == "1";
    let accrual: i64 = get_setting(conn, "loyalty_accrual_vnd_per_point")?
        .and_then(|v| v.parse().ok())
        .unwrap_or(10_000);
    let redeem: i64 = get_setting(conn, "loyalty_redeem_point_value")?
        .and_then(|v| v.parse().ok())
        .unwrap_or(1_000);
    Ok((enabled, accrual, redeem))
}
```

/// Tích điểm cho khách sau khi đơn được thanh toán.
/// - points = floor(order_total / accrual_vnd_per_point)
/// - Ghi point_transactions(ACCRUE) + cập nhật customers.points + customers.total_spent
/// - Nếu loyalty chưa bật HOẶC accrual_vnd_per_point <= 0 → không làm gì, trả AccrualResult { points_accrued: 0 }
/// - Không fail nếu cấu hình vắng (graceful degradation)
pub fn accrue_points_for_order(
    conn: &Connection,
    customer_id: i64,
    order_id: i64,
    order_total: i64,
) -> rusqlite::Result<AccrualResult> {
    let (enabled, accrual_vnd_per_point, _) = settings::get_loyalty_config(conn)?;
    if !enabled || accrual_vnd_per_point <= 0 {
        let current = get_customer(conn, customer_id)?;
        return Ok(AccrualResult {
            customer_id,
            points_accrued: 0,
            new_total_points: current.points,
        });
    }

    let points_accrued = order_total / accrual_vnd_per_point; // integer floor

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        if points_accrued > 0 {
            conn.execute(
                "INSERT INTO point_transactions(customer_id, order_id, kind, points, note)
                 VALUES (?1, ?2, 'ACCRUE', ?3, ?4)",
                params![
                    customer_id,
                    order_id,
                    points_accrued,
                    format!("Tích điểm đơn #{order_id}")
                ],
            )?;
        }
        conn.execute(
            "UPDATE customers
             SET points = points + ?2,
                 total_spent = total_spent + ?3
             WHERE id=?1",
            params![customer_id, points_accrued, order_total],
        )?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => {
            conn.execute_batch("COMMIT")?;
            let updated = get_customer(conn, customer_id)?;
            Ok(AccrualResult {
                customer_id,
                points_accrued,
                new_total_points: updated.points,
            })
        }
        Err(e) => {
            let _ = conn.execute_batch("ROLLBACK");
            Err(e)
        }
    }
}

// ─── Redeem ─────────────────────────────────────────────────────────────────

/// Đổi `points_to_redeem` điểm cho đơn `order_id`.
/// - Kiểm tra khách còn đủ điểm.
/// - discount_vnd = points_to_redeem × redeem_point_value
/// - Ghi point_transactions(REDEEM, -points_to_redeem) + trừ customers.points
/// - Trả về (discount_vnd: i64, new_total_points: i64)
pub fn redeem_points(
    conn: &Connection,
    customer_id: i64,
    order_id: i64,
    points_to_redeem: i64,
) -> rusqlite::Result<(i64, i64)> {
    let (_, _, redeem_point_value) = settings::get_loyalty_config(conn)?;
    let customer = get_customer(conn, customer_id)?;

    if points_to_redeem <= 0 {
        return Err(rusqlite::Error::InvalidParameterName(
            "Số điểm đổi phải lớn hơn 0".into(),
        ));
    }
    if customer.points < points_to_redeem {
        return Err(rusqlite::Error::InvalidParameterName(format!(
            "Không đủ điểm: có {} điểm, muốn đổi {} điểm",
            customer.points, points_to_redeem
        )));
    }

    let discount_vnd = points_to_redeem * redeem_point_value;

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        conn.execute(
            "INSERT INTO point_transactions(customer_id, order_id, kind, points, note)
             VALUES (?1, ?2, 'REDEEM', ?3, ?4)",
            params![
                customer_id,
                order_id,
                -points_to_redeem,
                format!("Đổi điểm đơn #{order_id}: -{points_to_redeem} điểm = -{discount_vnd} VND")
            ],
        )?;
        conn.execute(
            "UPDATE customers SET points = points - ?2 WHERE id=?1",
            params![customer_id, points_to_redeem],
        )?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => {
            conn.execute_batch("COMMIT")?;
            let updated = get_customer(conn, customer_id)?;
            Ok((discount_vnd, updated.points))
        }
        Err(e) => {
            let _ = conn.execute_batch("ROLLBACK");
            Err(e)
        }
    }
}

/// Phiên bản redeem có clamp: tự giới hạn points_to_redeem sao cho discount_vnd ≤ order_total.
/// Nếu points_to_redeem × redeem_point_value > order_total → giảm points_to_redeem = floor(order_total / redeem_point_value).
/// Trả về (discount_vnd: i64, new_total_points: i64).
pub fn redeem_points_clamped(
    conn: &Connection,
    customer_id: i64,
    order_id: i64,
    points_requested: i64,
    order_total: i64,
) -> rusqlite::Result<(i64, i64)> {
    let (_, _, redeem_point_value) = settings::get_loyalty_config(conn)?;
    if redeem_point_value <= 0 {
        return Err(rusqlite::Error::InvalidParameterName(
            "loyalty_redeem_point_value phải lớn hơn 0".into(),
        ));
    }
    // Clamp: tối đa bao nhiêu điểm để discount không vượt order_total
    let max_points_by_total = order_total / redeem_point_value;
    let customer = get_customer(conn, customer_id)?;
    let effective_points = points_requested
        .min(max_points_by_total)
        .min(customer.points);

    if effective_points <= 0 {
        return Ok((0, customer.points));
    }
    redeem_points(conn, customer_id, order_id, effective_points)
}

/// Điều chỉnh điểm thủ công (admin). delta có thể âm.
/// Không được để customers.points < 0.
pub fn adjust_points(conn: &Connection, input: AdjustPointsInput) -> rusqlite::Result<Customer> {
    let customer = get_customer(conn, input.customer_id)?;
    let new_points = customer.points + input.delta;
    if new_points < 0 {
        return Err(rusqlite::Error::InvalidParameterName(format!(
            "Điểm không thể âm: hiện có {} điểm, delta {}",
            customer.points, input.delta
        )));
    }

    conn.execute_batch("BEGIN")?;
    let result = (|| {
        conn.execute(
            "INSERT INTO point_transactions(customer_id, order_id, kind, points, note)
             VALUES (?1, NULL, 'ADJUST', ?2, ?3)",
            params![input.customer_id, input.delta, input.note.trim()],
        )?;
        conn.execute(
            "UPDATE customers SET points = points + ?2 WHERE id=?1",
            params![input.customer_id, input.delta],
        )?;
        Ok::<(), rusqlite::Error>(())
    })();
    match result {
        Ok(()) => {
            conn.execute_batch("COMMIT")?;
            get_customer(conn, input.customer_id)
        }
        Err(e) => {
            let _ = conn.execute_batch("ROLLBACK");
            Err(e)
        }
    }
}

    loyalty::redeem_points_clamped(
        &conn,
        customer_id,
        payload.order_id,
        payload.points_to_redeem,
        order_total,
    )
    .map_err(|e| e.to_string())
}

#[tauri::command]
pub fn adjust_points(db: State<AppDb>, payload: AdjustPointsInput) -> Result<Customer, String> {
    let conn = lock(&db)?;
    loyalty::adjust_points(&conn, payload).map_err(|e| e.to_string())
}

#[tauri::command]
pub fn list_point_transactions(
    db: State<AppDb>,
    customer_id: i64,
) -> Result<Vec<PointTransaction>, String> {
    let conn = lock(&db)?;
    loyalty::list_point_transactions(&conn, customer_id).map_err(|e| e.to_string())
}

import { Input } from "@/components/ui/input"

      </div>

    <div className="relative">
      <Input
        placeholder="Tìm khách theo SĐT / tên..."
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        onBlur={() => setTimeout(() => setOpen(false), 150)}
      />
      {open && (
        <ul className="absolute z-50 mt-1 w-full rounded-md border bg-background shadow-md">
          {results.slice(0, 5).map((c) => (
            <li
              key={c.id}
              className="cursor-pointer px-3 py-2 text-sm hover:bg-accent"
              onMouseDown={() => { onSelect(c); setQuery(""); setOpen(false) }}
            >
              <span className="font-medium">{c.name}</span>
              <span className="ml-2 text-muted-foreground">{c.phone}</span>
              <span className="ml-2 text-primary">{c.points} điểm</span>
            </li>
          ))}
        </ul>
      )}
    </div>
  )
}

import { useEffect, useState } from "react"
import { Link } from "react-router-dom"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Switch } from "@/components/ui/switch"
import { Dialog, DialogContent, DialogTitle, DialogTrigger } from "@/components/ui/dialog"
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs"
import { formatVnd } from "@/lib/format"
import {
  listCustomers, createCustomer, updateCustomer, setCustomerActive,
  adjustPoints, listPointTransactions,
  type Customer, type PointTransaction, type AdjustPointsInput,
} from "@/lib/api/customers"

export default function Customers() {
  const [customers, setCustomers] = useState<Customer[]>([])
  const [selected, setSelected] = useState<Customer | null>(null)
  const [transactions, setTransactions] = useState<PointTransaction[]>([])
  const [formName, setFormName] = useState("")
  const [formPhone, setFormPhone] = useState("")
  const [editingId, setEditingId] = useState<number | null>(null)
  const [dialogOpen, setDialogOpen] = useState(false)
  const [adjustDelta, setAdjustDelta] = useState(0)
  const [adjustNote, setAdjustNote] = useState("")
  const [error, setError] = useState<string | null>(null)

  async function refresh() {
    try { setCustomers(await listCustomers(true)) } catch (e) { setError(String(e)) }
  }

  async function loadTransactions(cid: number) {
    try { setTransactions(await listPointTransactions(cid)) } catch (e) { setError(String(e)) }
  }

  useEffect(() => { refresh() }, [])

  async function onSave() {
    setError(null)
    try {
      const payload = { name: formName.trim(), phone: formPhone.trim() }
      if (editingId != null) await updateCustomer(editingId, payload)
      else await createCustomer(payload)
      setDialogOpen(false); setFormName(""); setFormPhone(""); setEditingId(null)
      await refresh()
    } catch (e) { setError(String(e)) }
  }

  async function onToggle(c: Customer) {
    try { await setCustomerActive(c.id, !c.is_active); await refresh() } catch (e) { setError(String(e)) }
  }

  async function onAdjust(cid: number) {
    setError(null)
    if (!adjustNote.trim()) { setError("Ghi chú không được rỗng"); return }
    try {
      await adjustPoints({ customer_id: cid, delta: adjustDelta, note: adjustNote })
      setAdjustDelta(0); setAdjustNote("")
      await refresh()
      if (selected?.id === cid) await loadTransactions(cid)
    } catch (e) { setError(String(e)) }
  }

  function kindLabel(kind: string) {
    if (kind === "ACCRUE") return "Tích điểm"
    if (kind === "REDEEM") return "Đổi điểm"
    return "Điều chỉnh"
  }

  return (
    <div className="mx-auto flex max-w-5xl flex-col gap-4 p-6">
      <div className="flex flex-col gap-3 sm:flex-row sm:items-center sm:justify-between">
        <h1 className="text-2xl font-bold">Khách thân thiết</h1>
        <Button variant="outline" asChild><Link to="/">Về trang chủ</Link></Button>
      </div>

      {error && <p className="text-sm text-destructive">{error}</p>}

      <Tabs defaultValue="list">
        <TabsList className="h-11">
          <TabsTrigger className="px-4" value="list">Danh sách</TabsTrigger>
          <TabsTrigger className="px-4" value="history">Lịch sử điểm</TabsTrigger>
        </TabsList>

        {/* Tab 1: Danh sách khách */}
        <TabsContent value="list">
          <div className="flex flex-col gap-3 py-4">
            <div className="flex justify-end">
              <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
                <DialogTrigger asChild>
                  <Button onClick={() => { setEditingId(null); setFormName(""); setFormPhone("") }}>
                    Thêm khách
                  </Button>
                </DialogTrigger>
                <DialogContent>
                  <DialogTitle>{editingId != null ? "Cập nhật khách hàng" : "Thêm khách hàng mới"}</DialogTitle>
                  <div className="flex flex-col gap-3 pt-2">
                    <Input placeholder="Tên khách" value={formName} onChange={(e) => setFormName(e.target.value)} />
                    <Input placeholder="Số điện thoại" value={formPhone} onChange={(e) => setFormPhone(e.target.value)} />
                    {error && <p className="text-sm text-destructive">{error}</p>}
                    <div className="flex justify-end gap-2">
                      <Button variant="outline" onClick={() => setDialogOpen(false)}>Huỷ</Button>
                      <Button onClick={onSave}>Lưu</Button>
                    </div>
                  </div>
                </DialogContent>
              </Dialog>
            </div>

            <ul className="divide-y rounded-md border">
              {customers.map((c) => (
                <li key={c.id} className="flex items-center justify-between p-3">
                  <div>
                    <span className={`font-medium ${c.is_active ? "" : "text-muted-foreground line-through"}`}>
                      {c.name}
                    </span>
                    <span className="ml-2 text-sm text-muted-foreground">{c.phone}</span>
                    <span className="ml-2 text-sm font-semibold text-primary">{c.points} điểm</span>
                    <span className="ml-2 text-xs text-muted-foreground">
                      Tổng chi: {formatVnd(c.total_spent)}
                    </span>
                  </div>
                  <div className="flex items-center gap-3">
                    <Switch checked={c.is_active} onCheckedChange={() => onToggle(c)} />
                    <Button variant="ghost" size="sm" onClick={() => {
                      setEditingId(c.id); setFormName(c.name); setFormPhone(c.phone); setDialogOpen(true)
                    }}>Sửa</Button>
                    <Button variant="ghost" size="sm" onClick={() => {
                      setSelected(c); loadTransactions(c.id)
                    }}>Điểm</Button>
                  </div>
                </li>
              ))}
            </ul>
          </div>
        </TabsContent>

        {/* Tab 2: Lịch sử điểm */}
        <TabsContent value="history">
          <div className="flex flex-col gap-3 py-4">
            {selected ? (
              <>
                <div className="flex items-center justify-between rounded-md border p-3">
                  <div>
                    <span className="font-semibold">{selected.name}</span>
                    <span className="ml-2 text-muted-foreground">{selected.phone}</span>
                  </div>
                  <span className="text-lg font-bold text-primary">{selected.points} điểm</span>
                </div>

                {/* Điều chỉnh điểm thủ công */}
                <div className="flex items-center gap-2 rounded-md border p-3">
                  <Input
                    type="number"
                    placeholder="Delta điểm (+ hoặc -)"
                    value={adjustDelta}
                    className="w-36"
                    onChange={(e) => setAdjustDelta(Number(e.target.value))}
                  />
                  <Input
                    placeholder="Ghi chú bắt buộc"
                    value={adjustNote}
                    onChange={(e) => setAdjustNote(e.target.value)}
                  />
                  <Button onClick={() => onAdjust(selected.id)}>Điều chỉnh</Button>
                </div>

                {/* Lịch sử giao dịch */}
                <ul className="divide-y rounded-md border">
                  {transactions.map((t) => (
                    <li key={t.id} className="flex items-center justify-between px-3 py-2 text-sm">
                      <div className="flex items-center gap-2">
                        <span className={`font-medium ${t.points > 0 ? "text-green-600" : "text-red-600"}`}>
                          {t.points > 0 ? `+${t.points}` : t.points} điểm
                        </span>
                        <span className="text-muted-foreground">{kindLabel(t.kind)}</span>
                        {t.order_id && <span className="text-muted-foreground">đơn #{t.order_id}</span>}
                        {t.note && <span className="text-muted-foreground italic">{t.note}</span>}
                      </div>
                      <span className="text-xs text-muted-foreground">{t.created_at}</span>
                    </li>
                  ))}
                  {transactions.length === 0 && (
                    <li className="px-3 py-4 text-center text-sm text-muted-foreground">Chưa có giao dịch điểm nào</li>
                  )}
                </ul>
              </>
            ) : (
              <p className="py-8 text-center text-muted-foreground">Chọn khách từ tab "Danh sách" để xem lịch sử điểm.</p>
            )}
          </div>
        </TabsContent>
      </Tabs>
    </div>
  )
}
